PressedState toPressedState() {
  switch (this) {
    case SDL_PRESSED:
      return PressedState.pressed;
    case SDL_RELEASED:
      return PressedState.released;
    default:
      throw SdlError(this, 'Invalid pressed state.');
  }
}
